Than.js

Trailing heads method to adding vectors.


    // DEMO CODE
    const t1 = new Than({
      type: "polar",
      data: {
        theta: 45,
        magnitude: 5
      }
    });

    const t2 = new Than({
      type: "polar",
      data: {
        theta: -45,
        magnitude: 5
      }
    });

    const t3 = t1.add(t2);

    t3.log();
